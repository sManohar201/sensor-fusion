cmake_minimum_required(VERSION 3.16)
project(sensor_integration_lab_answer)
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(std_msgs REQUIRED)
find_package(sensor_msgs REQUIRED)
find_package(nav_msgs REQUIRED)
add_executable(frame_validator_answer src/frame_validator.cpp)
add_executable(sensor_health_hard_answer src/sensor_health_hard.cpp)
ament_target_dependencies(frame_validator_answer rclcpp std_msgs sensor_msgs nav_msgs)
ament_target_dependencies(sensor_health_hard_answer rclcpp std_msgs sensor_msgs nav_msgs)
install(TARGETS frame_validator_answer sensor_health_hard_answer DESTINATION lib/${PROJECT_NAME})
ament_package()
