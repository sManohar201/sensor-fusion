cmake_minimum_required(VERSION 3.16)
project(sensor_integration_lab)
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(std_msgs REQUIRED)
find_package(sensor_msgs REQUIRED)
find_package(nav_msgs REQUIRED)
add_executable(frame_validator src/frame_validator.cpp)
add_executable(sensor_health_hard src/sensor_health_hard.cpp)
ament_target_dependencies(frame_validator rclcpp std_msgs sensor_msgs nav_msgs)
ament_target_dependencies(sensor_health_hard rclcpp std_msgs sensor_msgs nav_msgs)
install(TARGETS frame_validator sensor_health_hard DESTINATION lib/${PROJECT_NAME})
ament_package()
